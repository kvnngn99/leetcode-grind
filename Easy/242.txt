class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        if len(s) != len(t):
            return False
        count = defaultdict(int)
        for s_char in s:
            count[s_char] += 1

        for t_char in t:
            count[t_char] -= 1

        for each_value in count.values():
            if each_value != 0:
                return False
        return True

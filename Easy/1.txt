class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        ##target - num = looking for that num in array
        for i in range(len(nums)):
            diff = target - nums[i]
            for (j) in range(i+1,len(nums)):
                if nums[j] == diff:

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        #Since we can only loop through nums once, cannot use nested forloops
        #Want to use Hashmap
        #The idea is we take the target number and subtract it by our current number to get a difference.
        #With this difference, we will loop through the for loop to check and see if that difference is there, if it is, then we return our current number index, and the difference number index

        myHashmap = {}
        
        for i in range(0, len(nums)):
            diff = target - nums[i]
            if diff in myHashmap:
                return [i, myHashmap[diff]]
            else:
                myHashmap[nums[i]] = i
        return -1 #no solutions found

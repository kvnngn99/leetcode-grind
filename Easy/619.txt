619. Biggest Single Number
12/19/2024
PostgreSQL
Sorting and Grouping

-- Remember subquery using WHERE, and COALESCE function
-- Write your PostgreSQL query statement below
SELECT
     COALESCE(MAX(num), NULL) AS num
FROM
    MyNumbers
WHERE num IN
    (SELECT
        num
    FROM
        MyNumbers
    GROUP BY
        num
    HAVING
        COUNT(num) = 1
    ORDER BY
        num DESC)
